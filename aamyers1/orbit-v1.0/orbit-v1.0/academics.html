A simple java program that runs the hailstone sequence based on a user input integer:
import java.util.*;
public class Extra01_06{
public static void main(String[] args) {
System.out.print("Enter a number: ");
Scanner sc = new Scanner(System.in);
int n = sc.nextInt();
int y;
int counter = 0;
while(n != 1) {
if(n % 2 == 0) {
y = n / 2;
System.out.println(n + " is even so I take half: " + y);
n = y;
counter ++;
}
else {
y = (3 * n) + 1;
System.out.println(n + " is odd so I make 3n+1: " + y);
n = y;
counter ++;
}
}
System.out.println("The process took " + counter + " steps to reach 1.");
}
}
